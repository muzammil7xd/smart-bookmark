# Local Development Environment Setup

# For local development, you need to set these environment variables.
# Copy this file to .env.local and fill in with your actual Supabase credentials.

# Supabase Project Settings (get from Supabase Dashboard → Settings → API)
NEXT_PUBLIC_SUPABASE_URL=https://your-project-id.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here

# To get these values:
# 1. Create a Supabase project at https://supabase.com
# 2. Go to Project Settings → API
# 3. Copy the "Project URL" and the "anon public" key
# 4. Paste them below, replacing the placeholder values

# After filling in these values, you can run:
# npm run dev
# 
# Then visit http://localhost:3000
